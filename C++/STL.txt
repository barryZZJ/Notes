哈希表用于自定义类型：
struct hasher{
  bool operator()(const E& e) const{return hash<string>()(e.string);}
};
struct cmp{
  bool operator()(const E&e1, const E&e2) const{return e1.string==e2.string;}
};
unordered_map<E, int, hasher, cmp>
