====== ProcessBuilder ======
Created 星期五 01 九月 2023

生成进程，获取输出

''import java.io.IOException;''

''public class RunPythonInVenv {''
''    public static void main(String[] args) {''
''        try {''
''            // Specify the path to the Python executable inside the virtual environment''
''            String venvPath = "pyvenv/Scripts/python.exe";''
            
''            // Create a ProcessBuilder to run the Python script with the argument''
''            ProcessBuilder processBuilder = new ProcessBuilder(venvPath, "your_python_script.py", "5000");''
            
''            // Redirect the output to the console''
''            processBuilder.redirectOutput(ProcessBuilder.Redirect.INHERIT);''
            
''            // Start the process''
''            Process process = processBuilder.start();''
            
''            // Wait for the process to finish (this step may be blocking)''
''            int exitCode = process.waitFor();''
            
''            // Check the exit code to determine if it was successful''
''            if (exitCode == 0) {''
''                System.out.println("Python script executed successfully.");''
''            } else {''
''                System.err.println("Python script execution failed with exit code: " + exitCode);''
''            }''
''        } catch (IOException | InterruptedException e) {''
''            e.printStackTrace();''
''        }''
''    }''
''}''
