Function makePara(isBold As Boolean) As Object
    With WordDoc.Paragraphs
      .Add
      Set Para = .Last
      With Para
        .Range.Style = StyleBill
        .FirstLineIndent = WordApp.CentimetersToPoints(0.85)
        .Range.ParagraphFormat.LineSpacingRule = 1 ' https://learn.microsoft.com/en-us/office/vba/api/publisher.pblinespacingrule
        .Range.Font.Bold = isBold
        .Range.Font.Size = 12 ' 小四
        .Range.Font.Name = "宋体"
        .Range.Font.Name = "Times New Roman"
        
      End With
    End With
    Set makePara = Para
End Function

Function round2(value As Double) As String
  round2 = Format(Round(value, 2), "0.00")
End Function

Function round4(value As Double) As String
  round4 = Format(Round(value, 4), "0.0000")
End Function

Function ColShift(ByVal offset As Integer) As String
    inputLetter = colStart
    Dim charCode As Integer
    Dim shiftedChar As String
    
    ' Check if inputLetter is a single character
    If Len(inputLetter) <> 1 Then
        ColShift = "Invalid Input"
        Exit Function
    End If
    
    ' Get the ASCII code of the input letter
    charCode = Asc(inputLetter)
    
    ' Check if the character is a letter (A-Z or a-z)
    If (charCode >= 65 And charCode <= 90) Or (charCode >= 97 And charCode <= 122) Then
        ' Shift the character code by the offset
        charCode = charCode + offset
        
        ' Check if the result is outside the A-Z or a-z range
        If (charCode < 65 And charCode > 90) Or (charCode < 97 And charCode > 122) Then
            ' Wrap around if necessary
            If charCode > 90 Then
                charCode = 64 + (charCode - 90)
            ElseIf charCode < 65 Then
                charCode = 91 + (charCode - 65)
            ElseIf charCode > 122 Then
                charCode = 96 + (charCode - 122)
            ElseIf charCode < 97 Then
                charCode = 123 + (charCode - 97)
            End If
        End If
        
        ' Convert the shifted character code back to a character
        shiftedChar = Chr(charCode)
        
        ColShift = shiftedChar
    Else
        ColShift = "Not a letter"
    End If
End Function

Function addDeviceMeasurementPara(Device As Integer, item As Integer) As String
    addDeviceMeasurementPara = ""
    rowDevice = (rowYear + Device * rowsPerDevice)
    If isBeijing Then
        '带month的变量对应的都是一个单行
        
        svrSystem = Range(ColShift(19) & rowDevice)
        svrSpec = Range(ColShift(20) & rowDevice)
        
        pwr = Range(ColShift(4) & rowDevice).value * 1000
        
        svrCnt = Range(ColShift(3) & rowDevice)
    Else
    
        svrSystem = Range(ColShift(10) & rowDevice)
        svrSpec = Range(ColShift(11) & rowDevice)
        
        pwr = Range(ColShift(5) & rowDevice).value * 1000
        
        svrCnt = Range(ColShift(2) & rowDevice)
    End If
    
    With makePara(isBold:=False)
      .Range.InsertAfter "（" & item & "）" & svrSystem & "，按照" & svrSpec & "功率测算，功耗为" & pwr & "瓦/台，共" & svrCnt & "台。"
      
      ' bold for svrSystem
      offset = Len("（" & item & "）")
      WordDoc.Range(.Range.Start + offset, .Range.Start + offset + Len(svrSystem)).Font.Bold = True
    
    End With
    
    
    If isBeijing Then
        With makePara(isBold:=False)
          .Range.InsertAfter "测算方法：" & pwr & "瓦/台*" & svrCnt & "台*（"
          WordDoc.Range(.Range.Start, .Range.Start + 4).Font.Bold = True
          
          For iMonth = 1 To 2
          
            rowMonth = (rowYear + Device * rowsPerDevice + iMonth - 1) ' for Col D
        
            monthstr = Range(ColShift(2) & rowMonth) '7月、8月 / 1-6月、9-12月
            
            runDays = Range(ColShift(6) & rowMonth)
                              
            
            runHours = Range(ColShift(5) & rowMonth).value
            
            runHoursPeak = Range(ColShift(7) & rowMonth).value
            runHoursHigh = Range(ColShift(9) & rowMonth).value
            runHoursNormal = Range(ColShift(11) & rowMonth).value
            runHoursLow = Range(ColShift(13) & rowMonth).value
            
            priceMonth = round2(Range(ColShift(15) & rowMonth))
            pricePeak = round4(Range(ColShift(8) & rowMonth))
            priceHigh = round4(Range(ColShift(10) & rowMonth))
            priceNormal = round4(Range(ColShift(12) & rowMonth))
            priceLow = round4(Range(ColShift(14) & rowMonth))
          
            
            
            .Range.InsertAfter "（"
            If iMonth = 1 Then
              .Range.InsertAfter runHoursPeak & "小时/天*" & pricePeak & "元/千瓦时+"

              .Range.InsertAfter "+"

        End With
    Else

        rowMonth = (rowYear + Device * 1 + iMonth - 1) ' for Col D

Function addSumPerYearPara() As String

  If isBeijing Then
    With makePara(isBold:=False)

        rowDevice = (rowYear + Device * rowsPerDevice)
        .Range.InsertAfter round2(Range(ColShift(16) & rowDevice)) & "万元+"
      Next Device

      .Range.InsertAfter round2(Range(ColShift(16) & rowDevice)) & "万元"
      .Range.InsertAfter "=" & round2(Range(ColShift(17) & rowYear)) & "万元。"
    End With
  Else
    With makePara(isBold:=False)
      .Range.InsertAfter "（1）至（" & deviceCnt & "）燃料动力费小计："
      
      For Device = 0 To deviceCnt - 1
        rowDevice = (rowYear + Device * rowsPerDevice)
        .Range.InsertAfter round2(Range(ColShift(7) & rowDevice)) & "万元"
        If Device <> deviceCnt - 1 Then
          .Range.InsertAfter "+"
        End If
      Next Device
      .Range.InsertAfter "=" & round2(Range(ColShift(8) & rowYear)) & "万元。"
    End With
  End If
End Function

Sub makeEBill()
    ' Create a new Word application
    Set WordApp = CreateObject("Word.Application")
    
    ' Add a new document
    Set WordDoc = WordApp.Documents.Add
    
    Set StyleBill = WordDoc.Styles.Add(Name:="Bill")
    
    ' ===================== arguments
    isBeijing = Range("A7").value = "Y"
    deviceCnt = Range("C7")
    
    rowStart = Range("B7")
    colStart = Range("G7")
    rowsPerDevice = Range("D7")
    
    sysName = Range("E7")
    companyName = Range("F7")
    
    ' ====================
    
    'With makePara(isBold:=False)
      '.Range.InsertAfter "改动：行距1.5；"
    'End With
    
    
    ' header
    With makePara(isBold:=True)
      .Range.InsertAfter "2.3.x " & companyName & "燃料动力费"
     
    End With
    
    For iYear = 1 To 3
      
      With makePara(isBold:=True)
      
        rowYear = rowStart + (iYear - 1) * (deviceCnt * rowsPerDevice)
        
        yearstr = Range(ColShift(0) & rowYear) '第X年
      
        .Range.InsertAfter iYear & "）课题执行期" & yearstr
        
      End With
    
      rowComputerPerYear = rowYear + (deviceCnt - 1) * rowsPerDevice
      If isBeijing Then
        runDaysPerServer(1, iYear) = Range(ColShift(6) & rowYear).value
        runDaysPerServer(2, iYear) = Range(ColShift(6) & (rowYear + 1)).value
        runDaysPerServer(3, iYear) = runDaysPerServer(1, iYear) + runDaysPerServer(2, iYear)
        
        runDaysPerComputer(1, iYear) = Range(ColShift(6) & rowComputerPerYear).value
        runDaysPerComputer(2, iYear) = Range(ColShift(6) & (rowComputerPerYear + 1)).value
        runDaysPerComputer(3, iYear) = runDaysPerComputer(1, iYear) + runDaysPerComputer(2, iYear)
        
        runHoursPerServer(1, iYear) = Range(ColShift(5) & rowYear).value
        runHoursPerServer(2, iYear) = Range(ColShift(5) & (rowYear + 1)).value
        
        runHoursPerComputer(1, iYear) = Range(ColShift(5) & rowComputerPerYear).value
        runHoursPerComputer(2, iYear) = Range(ColShift(5) & (rowComputerPerYear + 1)).value
      Else
        runDaysPerServer(3, iYear) = Range(ColShift(3) & rowYear).value
        runDaysPerComputer(3, iYear) = Range(ColShift(3) & rowComputerPerYear).value
        
        runHoursPerServer(1, iYear) = Range(ColShift(4) & rowYear).value
        runHoursPerComputer(1, iYear) = Range(ColShift(4) & rowComputerPerYear).value
      End If
      
      addHeaderFirstPara iYear
      
      Dim item As Integer
      
      addHeaderSecondPara iYear
      
      'Dim priceDevicePerMonth() As String
      'ReDim priceDevicePerMonth(0 To deviceCnt - 1) As String
      
      item = 1
      Dim Device As Integer
      
      For Device = 0 To deviceCnt - 1

        addDeviceMeasurementPara Device, item
        
        'priceDevicePerMonth(Device) = round2(Range("R" & rowMonth).value)
        
        item = item + 1
        
      Next Device
      
      
      addSumPerYearPara
    Next iYear
    
    addSumTotalPara
    
    With makePara(isBold:=True)
      .Range.InsertAfter "4）" & companyName & "燃料动力费汇总"
    End With
    
    ' Make Word visible (optional)
    WordApp.Visible = True
    
    ' Release the Word objects
    Set WordDoc = Nothing
    Set WordApp = Nothing
End Sub
