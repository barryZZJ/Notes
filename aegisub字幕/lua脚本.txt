re.match(str,pat) re.find(str,pat)
	find: 匹配多次，不捕获subgroup。返回match table数组，每个元素是一个匹配结果的**match table**
	match: 只匹配一次，捕获subgroup。返回match table数组，[1]是整个匹配结果的**match table**，后续的是捕获的subgroup的**match table**
	无匹配则返回nil

{{{code: lang="lua" linenumbers="False"
function process_lines(subtitles, selected_lines, active_line)
	sels = {}
	local i_dialogue = 1

	for _, i in ipairs(selected_lines) do  -- or for i = 1, #subtitles
		aegisub.progress.set(i * 100 / #subtitles)

		local l = subtitles[i]  # subtitle object
		if subtitles[i].class == "dialogue" then
			aegisub.debug.out(string.format('Processing line %d: "%s"\n', i_dialogue, l.text))

			nl = l
			subtitles[i] = l
			table.insert(sels, i)
		    i_dialogue = i_dialogue + 1
		end
	end
	aegisub.set_undo_point(script_name)
	return sels
end
}}}
