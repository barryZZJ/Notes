2. 'beforescriptexecute' event is HTML5 specifications, is supported only by Gecko-based browsers (Firefox) at the moment.
element.onbeforescriptexecute
当 HTML 文档中的<script>标签内的代码将要执行时触发该事件，如果这个script标签是用appendChild()等方法动态添加上去的，则不会触发该事件。

var changed = 0; // script need to be edited with

window.addEventListener('beforescriptexecute', function(e) {

    ///for external script:
	src = e.target.src;
	if (src.search(/bad\.js/) != -1) {
                changed++;
		e.preventDefault();
		e.stopPropagation();
		append(NewScript1);
	};

    ///for inline script:
        if(e.target===document.getElementsByTagName("script")[0]){
            changed++;
            e.stopPropagation();
            e.preventDefault();

        }
        //tips: you could also run a regex search for the e.target.innerHTML

	if(changed == 2) window.removeEventListener(e.type, arguments.callee, true);
