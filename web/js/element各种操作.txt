getElementByText 自定义：
{{{code: lang="js" linenumbers="False"
//未找到则返回空
function getElementByText( text, node ){
  return document.evaluate("//*[.='"+text+"']", 
	 node || document, null, XPathResult.ANY_TYPE, null).iterateNext();
}
var myElement = getElementByText( "This is the string I need to match" );
}}}

或
{{{code: lang="js" linenumbers="False"
function findByText(node, text) {
    if(node.nodeValue == text) {
        return node.parentNode;
    }

    for (var i = 0; i < node.childNodes.length; i++) {
        var returnValue = findByText(node.childNodes[i], text);
        if (returnValue != null) {
            return returnValue;
        }
    }

    return null;
}
}}}


CREATE:
document.createElement(String) - "div" "p" 等
document.createElement('div');
封装一个更方便的：
{{{code: lang="js" linenumbers="False"
function createElement(element, attribute, inner) {
  if (typeof(element) === "undefined") {
	return false;
  }
  if (typeof(inner) === "undefined") {
	inner = "";
  }
  var el = document.createElement(element);
  if (typeof(attribute) === 'object') {
	for (var key in attribute) {
	  el.setAttribute(key, attribute[key]);
	}
  }
  if (!Array.isArray(inner)) {
	inner = [inner];
  }
  for (var k = 0; k < inner.length; k++) {
	if (inner[k].tagName) {
	  el.appendChild(inner[k]);
	} else {
	  el.appendChild(document.createTextNode(inner[k]));
	}
  }
  return el;
}

}}}

用例：
''createElement("tag",{id:val,class:val},"inner text");''
''createElement("tag",{id:val,class:val},[element1,"some text",element2,element3]);''

APPEND:
1.elem.append(DOMString or elem) 可以有多个参数，向匹配元素集合中的每个元素结尾插入由参数指定的内容。DOMString = "p" / "div" 等

2.elem.appendChild(elem) 只能传一个节点，且不直接支持传字符串，返回追加的 Node 节点；
若 appendChild() 的参数是页面存在的一个元素，则执行后原来的元素会被移除；

3.elem.appendTo(par) 向目标结尾插入集合中的每个元素。
elem.appendTo('body') // 好像是jq的功能
