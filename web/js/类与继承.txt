2. class方法声明
https://developer.mozilla.org/zh-CN/docs/Web/JavaScript/Reference/Statements/class
{{{code: lang="js" linenumbers="False"
class Point {
  constructor(x, y) {
	this.x = x;
	this.y = y;
  }
  
  static Name = "hello";

  something() {
	return '(' + this.x + ', ' + this.y + ')';
  }
  static create(x, y){
  	return new Point(x, y);
  }
}
var p = new Point(1,2);
}}}

继承：
{{{code: lang="js" linenumbers="False"
// Define the superclass
class Animal {
  constructor(name) {
    this.name = name;
  }

  speak() {
    console.log(`${this.name} makes a noise.`);
  }
}

// Define the subclass
class Dog extends Animal {
  constructor(name, breed) {
    super(name);
    this.breed = breed;
  }

  speak() {
    console.log(`${this.name} barks.`);
  }
}

// Create an instance of the superclass
const animal = new Animal('Animal');

// Call the superclass method
animal.speak(); // "Animal makes a noise."

// Create an instance of the subclass
const dog = new Dog('Dog', 'Labrador');

// Call the subclass method
dog.speak(); // "Dog barks."

// Access properties inherited from the superclass
console.log(dog.name); // "Dog"

}}}
