====== 获取管理员权限 ======
Created 星期三 11 九月 2024


NET SESSION >nul 2>&1
IF %ERRORLEVEL% EQU 0 (
	echo Running with admin privileges
) ELSE (
	echo Running without admin privileges
	echo Requesting admin privileges...
	powershell Start-Process -Verb RunAs -FilePath "%0"
	exit /B
)


执行cmd命令：
powershell -Command "Start-Process cmd -ArgumentList '/c xxx' -Verb RunAs"
