os.path

.splitext(filepath:文件路径)：-> (文件所在目录+文件名, .后缀名)
.splitext(path:目录路径)：-> (目录路径, 空)
.split(filepath:文件路径) -> (文件目录（不带结尾/）, 文件名带后缀)
.split(filepath:目录路径) -> (上一级目录（不带结尾/）, 最后一级目录名）
.basename(path) 文件名带后缀 或 最后一级路径
.dirname(path) 文件所在目录 或 父目录

os.sep = "\\"

#只遍历根目录下所有目录和文件：
scandir返回生成器
for entry in os.scandir(path):
	entry: os.DirEntry
	entry.name # 只有文件名和后缀名
	entry.path # 完整目录带文件和后缀名
	entry.is_dir()
	entry.is_file()

#递归遍历目录下的所有目录和文件：
（walk返回生成器，可以用next遍历，第一个就是根目录下的信息）
for root,dirs,files in os.walk(r"D:\test"):
	for file in files:
		#获取文件所属目录
		print(root)
		#获取文件路径
		print(os.path.join(root,file))
#获取test目录下的所有目录
	for dir in dirs:
		#获取目录的名称
		print(dir)
		#获取目录的路径
		print(os.path.join(root,dir))

os.listdir(path) 返回一个数组，只包含根目录下所有目录和文件名，随机顺序。

指定遍历顺序：sorted()
numbers = re.compile(r'(\d+)')
def numericalSort(value):
	parts = numbers.split(value)
	parts[1::2] = map(int, parts[1::2])
	return parts
