====== Logging ======
Created 星期六 22 十月 2022

import logging
https://docs.python.org/3/library/logging.html

格式：
https://docs.python.org/3/library/logging.html#logrecord-attributes
https://docs.python.org/3/library/logging.html#logging.basicConfig

.debug
.info
.warning
.error

只支持一个参数

默认level为WARNING

设置logging默认配置（也作用于logger）：
fmt = '%(asctime)s [%(levelname)s] %(message)s'
datefmt = '%Y-%m-%d %H:%M:%S'
logging.basicConfig(format=fmt, datefmt=datefmt, level=logging.INFO)

设置logger配置：
log = logging.getLogger(name: Optional)
log.setLevel(logging.INFO)  # 需要先调用logging.basicConfig() （不用带参数）才能生效

输出到文件
logging.basicConfig(filename='example.log', encoding='utf-8', level=logging.DEBUG)

handlers:
https://docs.python.org/3/howto/logging.html#handlers
可以单独设置一个handler的format:
fmter = logging.Formatter(fmt=fmt, datefmt=datefmt)
hdlr = logging.FileHandler(PATHS.LOGFILE, encoding='utf8')
hdlr.setFormatter(fmter)
logger.addHandler(hdlr)

同时输出到stdout和文件：
{{{code: lang="python3" linenumbers="False"
logging.basicConfig(
    level=logging.DEBUG,
    format="%(asctime)s [%(levelname)s] %(message)s",
    datefmt='%Y-%m-%d %H:%M:%S',
    handlers=[
        logging.FileHandler(f"log_{datetime.today().strftime('%Y-%m-%d_%H-%M-%S')}.log", encoding='utf8'),# 可以设置mode='a'
        logging.StreamHandler()  # 默认输出到sys.stderr，如有特殊需求需手动指定sys.stdout
    ]
)
# or
logger = logging.getLogger()
logger.setLevel(logging.DEBUG)

output_file_handler = logging.FileHandler(f"log_{datetime.today().strftime('%Y-%m-%d_%H-%M-%S')}.log")
stdout_handler = logging.StreamHandler(sys.stdout)

logger.addHandler(output_file_handler)
logger.addHandler(stdout_handler)
}}}
