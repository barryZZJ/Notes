import sys 

def debugger_is_active() -> bool:
	"""Return if the debugger is currently active"""
	return hasattr(sys, 'gettrace') and sys.gettrace() is not None

DEBUG = hasattr(sys, 'gettrace') and sys.gettrace() is not None
