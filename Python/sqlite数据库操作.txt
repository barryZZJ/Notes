====== sqlite数据库操作 ======
Created 星期日 11 九月 2022

https://www.runoob.com/sqlite/sqlite-python.html

import sqlite3

conn = sqlite3.connect("database.db")
c = conn.cursor()

# CREATE
c.execute('CREATE TABLE IF NOT EXISTS numbers (id integer primary key ,number varchar(20) NOT NULL)') 
conn.commit()

# SELECT
cursor = c.execute("SELECT id, name, address, salary  from COMPANY")
for row in cursor: # 无论field的输入，都返回元组
   print "ID = ", row[0]
   print "NAME = ", row[1]
   print "ADDRESS = ", row[2]
   print "SALARY = ", row[3], "\n"

conn.close()
