from docx import Document

def write_to_word(title, content, filepath):
	# Create a new Document
	doc = Document()
	# Add a title
	doc.add_heading(title, level=1)
	# Add the content
	doc.add_paragraph(content)
	# Save the document
	doc.save(filepath)

# Example usage
title = "Sample Title"
content = "This is the sample content for the Word document."
filepath = "sample_document.docx"
write_to_word(title, content, filepath)
