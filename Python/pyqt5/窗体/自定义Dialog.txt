====== 自定义Dialog ======
Created 星期二 22 十月 2024

dialog = QDialog(self, Qt.WindowSystemMenuHint | Qt.WindowTitleHint | Qt.WindowCloseButtonHint)
dialog.setWindowTitle
bbox = QDialogButtonBox(QDialogButtonBox.Ok | QDialogButtonBox.Cancel, dialog)
bbox.accepted.connect
bbox.rejected.connect

.show() 非阻塞
.exec() 阻塞
