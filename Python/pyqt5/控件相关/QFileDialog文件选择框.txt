import sys
import os
from PyQt5.QtWidgets import QWidget, QFileDialog
from PyQt5.QtWidgets import QApplication

class FilePicker(QFileDialog):
    def __init__(self, parent: QWidget=None):
        super(FilePicker, self).__init__(parent)

    def pickVideo(self, dir=os.getcwd()) -> str:
        self.setFileMode(QFileDialog.ExistingFile)
        self.setDirectory(dir)
        self.setNameFilter('*.mkv *.mp4 *.flv *.avi *.wmv *.mpg *.avs')
        if self.exec_():
            # 选中文件, 返回码==1
            # return list of full path
            filenames = self.selectedFiles()
            return filenames[0].replace('/', '\\')
        # 未选中文件，返回码==0
        return ''

    def pickFiles(self, dir=os.getcwd()) -> list[str]:
        self.setFileMode(QFileDialog.ExistingFiles)
        self.setDirectory(dir)
        if self.exec_():
            # 选中文件, 返回码==1
            # return list of full path
            filenames = self.selectedFiles()
            return [filename.replace('/', '\\') for filename in filenames]
        # 未选中文件，返回码==0
        return []
