=== msgbox: ===
https://blog.csdn.net/jia666666/article/details/81540785
from PyQt5.QtWidgets import QMessageBox

QMessageBox.information(None, None, "Debug mode is on")
(parent, title, text, button, defaultButton)
.critical 
.warning 
.question
（不能按参数名传参）

图标会跟随app.setWindowIcon（或者parent的？不确定），不需要单独设置

定时关闭：
msg_box = QMessageBox()
msg_box.setIcon(QMessageBox.Information)
msg_box.setWindowTitle("Notification")
msg_box.setText("This message will close in 3 seconds.")
msg_box.setStandardButtons(QMessageBox.Ok)
QTimer.singleShot(3000, msg_box.close)

包装后：
''from PyQt5.QtCore import QTimer''
''from PyQt5.QtWidgets import QMessageBox''


''class QTimedMessageBox(QMessageBox):''
	''def __init__(self, timeout, *args, **kwargs):''
		''self.timeout = timeout''
		''super().__init__(*args, **kwargs)''

	''def showEvent(self, event):''
		''QTimer.singleShot(self.timeout, self.close)''

	''@staticmethod''
	''def information(parent, title, text, buttons=QMessageBox.Ok, defaultButton=QMessageBox.Ok, *, timeoutms):''
		''msg_box = QMessageBox()''
		''msg_box.setIcon(QMessageBox.Information)''
		''msg_box.setWindowTitle(title)''
		''msg_box.setText(text)''
		''msg_box.setStandardButtons(buttons)''
		''msg_box.setDefaultButton(defaultButton)''
		''QTimer.singleShot(timeoutms, msg_box.close)''
		''return msg_box.exec_()''
