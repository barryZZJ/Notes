from openpyxl import load_workbook, workbook
# read
wb = load_workbook('input.xlsx', read_only=False)  # type: workbook.Workbook
wb.sheetnames  # ['Sheet1', 'Sheet2']
# worksheet
ws = wb.active  # default is first sheet
ws = wb['Sheet1']
for sheet in wb:
    print(sheet.title)
# read contents
for row in ws.iter_rows(min_row=1, max_col=3, max_row=2, values_only=False):
	for cell in row:
       print(cell)  # Cell. value if values_only=True
ws.rows  # list[list[Cell]]

for col in ws.iter_cols(min_row=1, max_col=3, max_row=2, values_only=False):
    for cell in col:
        print(cell)  # Cell
ws.columns  # list[list[Cell]]

for row in ws.values:
   for value in row:
     print(value)
     
# save
wb.save('filename.xlsx')

自动设置列宽：（如果index=False则set_column时不用+1）
def adjust_col_width(df, f, sheet_name, has_index=True):
	worksheet = f.sheets[sheet_name]  # pull worksheet object
	for idx, col in enumerate(df):  # loop through all columns
		series = df[col]
		max_len = max((
			series.astype(str).map(len).max(),  # len of largest item
			len(str(series.name))  # len of column name/header
		))+1  # adding a little extra space
		offset = 1 if has_index else 0
		worksheet.set_column(idx + offset, idx + offset, max_len)  # set column width

def put_headers(df, f, sheetname):
	workbook = f.book
	worksheet = f.sheets[sheetname]
	header_format = workbook.add_format({
		'bold': False,
		'align': 'right',
		"font_name": "Times New Roman",
	})

	for col, value in enumerate(df.columns.values):
		worksheet.write(0, col, value, header_format)

def set_col_format(df, f, sheetname):
	worksheet = f.sheets[sheetname]
	fmt = f.book.add_format({
		'align': 'right',
		'font_name': "Times New Roman",
		'num_format': '0.000',
	})
	worksheet.set_column(0, len(df), None, fmt)
