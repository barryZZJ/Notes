{{{code: lang="python3" linenumbers="False"
import multiprocessing

def callback_function(value):
    print("Callback function executed with value:", value)

def register_callback(queue):
    # Get the function from the queue
    callback_func = queue.get()
    # Invoke the callback function with a value
    callback_func("Hello from the registered callback!")

if __name__ == '__main__':
    # Create a multiprocessing.Queue to pass the callback function
    queue = multiprocessing.Queue()
    # Create a Process object that executes the register_callback function
    process = multiprocessing.Process(target=register_callback, args=(queue,))
    # Start the process
    process.start()
    # Put the callback_function into the queue
    queue.put(callback_function)
    # Wait for the process to finish
    process.join()

}}}
