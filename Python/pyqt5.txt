trigger = QtCore.pyqtSignal(str) # 自定义信号对象，要定义成静态成员。参数str就代表这个信号可以传一个字符串
self.trigger.emit(str(i)) #触发自定义信号
self.work.trigger.connect(self.display) # 把信号连接到外部的槽函数
注意QThread对象要用self保存，否则外层函数结束了这个thread对象就被删除了

简单模板：
import sys
from PyQt5 import QtCore, QtGui, QtWidgets
from PyQt5.QtWidgets import QApplication, QDialog, QMessageBox
from PyQt5.QtCore import Qt

import gui_file_name

if __name__ == '__main__':
	app = QApplication(sys.argv)
	MainWindow = QDialog() # 或QMainWindow()、QWidget()等等
	ui = gui_file_name.Ui_MainWindow()
	ui.setupUi(MainWindow)
	ui.retranslateUi(MainWindow)
	MainWindow.setFixedSize(MainWindow.width(), MainWindow.height())
	MainWindow.setWindowFlags(Qt.WindowMinimizeButtonHint|Qt.WindowCloseButtonHint)
	MainWindow.show()
	sys.exit(app.exec_()) # 如果有多线程，需要os._exit(app.exec_())

带事件函数的高级模板：
import sys
from PyQt5.QtCore import Qt, pyqtSlot
from PyQt5.QtWidgets import QApplication, QDialog, QWizard, QWizardPage

from wizard import Ui_Wizard


class MyWizard(QWizard):
	def __init__(self):
		super(MyWizard, self).__init__()
		Ui_Wizard().setupUi(self)
		self.setFixedSize(self.width(), self.height())
		self.setWindowFlags(Qt.WindowMinimizeButtonHint)

	@QtCore.pyqtSlot() #如果有参数的话，装饰器pyqtSlot()里必须要加上参数类型
	def on_pushButton_clicked(self):
	#命名格式on_Child_Signal，主窗口必须手动连接槽函数
		print('button clicked!')


app = QApplication(sys.argv)
MainWindow = MyWizard()
MainWindow.show()
sys.exit(app.exec_()) # 如果有多线程，需要os._exit(app.exec_())

手动连接事件函数：
signal有.connect方法连接到槽函数
一：
designer中不设置，在代码中写：（如果信号带多个参数，对应槽函数包含多个参数即可）
def connectSlots(ui: Ui_Form):
	ui.xxButton.clicked.connect(lambda ui: func(ui)) 或实现函数闭包，.connect(fun(ui))中fun(ui)返回一个函数即可
xxButton.clicked.connect(funcname) def funcname(txt): ...
self.connect(self.man_btn, QtCore.SIGNAL('clicked()'), self.manual_slot)

二：
designer中选择设置信号/槽，把按钮连接到form上，在form那一栏里编辑，添加槽函数。
新建QDialog的子类，实现刚刚添加的槽函数
self.chk_modsw.stateChanged['int'].connect(Form.stateChanged) # 0:no, 1:half, 2:check

信号与槽函数：
from PyQt5.QtCore import pyqtSignal, pyqtSlot, QObject
class Obj(QObject):
	newSignal = pyqtSignal(<type>) # 如果信号有参数，type里指明参数类型，如str, int
	def doSomething(self):
		....
		self.newSignal.emit(<param>) # 发射信号
连接槽函数
1. obj.newSignal.connect(slotFunc) # 槽函数捕捉信号即可
2. Obj(newSignal=slotFunc) # 初始化时直接传递槽函数也可
