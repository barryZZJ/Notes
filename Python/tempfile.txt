====== tempfile ======
Created 星期三 20 十二月 2023

%temp% 临时文件夹
temp_dir = os.getenv('TEMP')

TemporaryFile 没有文件名
NamedTemporaryFile 有文件名，通过.name获取

import tempfile

fp = tempfile.TemporaryFile()
fp.write(b'Hello world!')
fp.seek(0)
fp.read()
fp.close()

# 在退出上下文管理器时，临时文件会自动关闭和删除
with tempfile.TemporaryFile() as temp_file:
	temp_file.write(b"Hello, World!")
