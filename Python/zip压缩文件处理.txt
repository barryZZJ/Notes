import zipfile
压缩：
with zipfile.ZipFile(zip文件名, "w", zipfile.ZIP_DEFLATED) as zf:
    zf.write(要压缩的文件名, 压缩包中的文件名)

解压：
with zipfile.ZipFile(zip文件名) as zf:
    zf.extractall(dir=None)
