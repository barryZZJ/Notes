====== nginx ======
Created 星期日 03 九月 2023

可以作为反向代理/反代（转发流量）


# https://www.nginx.com/resources/wiki/start/
# https://www.nginx.com/resources/wiki/start/topics/tutorials/config_pitfalls/
# https://wiki.debian.org/Nginx/DirectoryStructure

sudo apt install nginx
sudo sysytemctl start nginx
sudo systemctl enable nginx

configure:
把静态文件拷到/var/www/，要不然可能没有访问权限
sudo chown -R www-data:www-data /var/www/yacd
sudo chmod -R 755 /var/www/yacd
nano /etc/nginx/sites-available/barryzzj.top: （每个{server}块对应一个服务，每个文件对应一个域名）

基于/<path>的反代：（有点鸡肋，大部分web app都基于/path的格式，就找不到了。只能考虑配合frpc的http代理功能，支持自定义域名）
''server {''
	''listen 80;''
	''server_name yourdomain.com;''

	''location / {''
		''root /var/www/yacd;''
		''try_files $uri $uri/ /index.html;''
	''}''
''}''

基于 server_name 的反代（不同域名跳转到不同服务）：
''# App1 - routes app1.mydomain.com to backend service 1''
''server {''
	''listen 80;''
	''server_name app1.mydomain.com;''
    
	''location / {''
		''proxy_pass http://localhost:3001;  # or your app1 service URL''
		''proxy_set_header Host $host;''
		''proxy_set_header X-Real-IP $remote_addr;''
		''proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;''
		''proxy_set_header X-Forwarded-Proto $scheme;''
	''}''
''}''

''# Optional: Default server for unmatched hosts''
''server {''
	''listen 80 default_server;''
	''server_name _;''
	''return 444;  # Close connection without response''
''}''

sudo ln -s /etc/nginx/sites-available/barryzzj.top /etc/nginx/sites-enabled/
sudo nginx -t
sudo systemctl reload nginx

查看log
sudo tail -f /var/log/nginx/access.log
sudo tail -f /var/log/nginx/error.log
